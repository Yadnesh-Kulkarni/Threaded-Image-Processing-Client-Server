cmake_minimum_required(VERSION 3.13)
project(EOS_Project)
find_package( OpenCV REQUIRED )
set(CMAKE_CXX_STANDARD 14) 
#set(CMAKE_BUILD_TYPE Debug)
set(UTILS_FILES ./../src/Utils/csapp.c ./../src/Utils/cvutils.cpp)
include_directories( ${OpenCV_INCLUDE_DIRS} ../src/include)
set(CLIENT_SRC ./../src/Client/client.c ${UTILS_FILES})
add_executable( client ${CLIENT_SRC})
set_target_properties(client PROPERTIES RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/../")
set(SERVER_SRC ./../src/Server/server.c ./../src/Server/queue.c ./../src/Server/handleclient.c ${UTILS_FILES})
add_executable( server ${SERVER_SRC})
set_target_properties( server PROPERTIES RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/../")
target_link_libraries( server ${OpenCV_LIBS} )
target_link_libraries( client ${OpenCV_LIBS} )
